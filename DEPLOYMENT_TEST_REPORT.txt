```json
      "src": "/api/backend/(.*)",
      "dest": "/backend/src/index.ts"
    },
    {
      "src": "/api/(.*)",
      "dest": "/next-app/api/$1"
    },
    {
      "src": "/(.*)",

    }
  ]
}
```
